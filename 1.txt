#region 批量插入
        /// <summary>
        /// 批量插入数据
        /// </summary>
        /// <param name="list"></param>
        /// <returns></returns>
        public bool BulkCopy(List<WZCityConfigEntity> list)
        {
            bool result = false;

            try
            {
                #region 组织table
                DataTable dt = new DataTable();

                dt.Columns.Add("CityID", typeof(int));
                dt.Columns.Add("ParentID", typeof(int));
                dt.Columns.Add("CityName", typeof(string));
                dt.Columns.Add("ApiKey", typeof(string));

                dt.Columns.Add("Ecode", typeof(int));
                dt.Columns.Add("Vcode", typeof(int));
                dt.Columns.Add("Owner", typeof(char));
                dt.Columns.Add("IdNum", typeof(char));

                dt.Columns.Add("RegCertCode", typeof(char));
                dt.Columns.Add("RegPhone", typeof(char));
                dt.Columns.Add("IsChild", typeof(char));

                dt.Columns.Add("CreateTime", typeof(DateTime));
                dt.Columns.Add("UpdateTime", typeof(DateTime));
                dt.Columns.Add("IsValid", typeof(char));
                dt.Columns.Add("CityOrder", typeof(int));
                dt.Columns.Add("ShowName", typeof(string));

                foreach (WZCityConfigEntity entity in list)
                {
                    DataRow dr = dt.NewRow();

                    dr["CityID"] = entity.CityID;
                    dr["ParentID"] = entity.ParentID;
                    dr["CityName"] = entity.CityName;
                    dr["ApiKey"] = entity.ApiKey;

                    dr["Ecode"] = entity.Ecode;
                    dr["Vcode"] = entity.Vcode;
                    dr["Owner"] = entity.Owner;
                    dr["IdNum"] = entity.IdNum;

                    dr["RegCertCode"] = entity.RegCertCode;
                    dr["RegPhone"] = entity.RegPhone;
                    dr["IsChild"] = entity.IsChild;

                    dr["CreateTime"] = entity.CreateTime;
                    dr["UpdateTime"] = entity.UpdateTime;

                    dr["IsValid"] = entity.IsValid;
                    dr["CityOrder"] = 100;
                    dr["ShowName"] = entity.CityName;

                    dt.Rows.Add(dr);
                }
                #endregion

                result = DbHelperSQL.BulkCopy(dt, "WZCityConfig");
            }
            catch (Exception ex)
            {
                LogHelp.Error("车轮城市新增数据异常", ex);
            }

            return result;
        }
        #endregion
